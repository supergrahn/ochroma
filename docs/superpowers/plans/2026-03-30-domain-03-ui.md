```rust
// crates/vox_ui/src/vello_ctx.rs
//! VelloCtx — wgpu-backed Vello scene renderer for game UI.
//!
//! VelloCtx owns a vello::Renderer and accumulates draw commands each frame.
//! VelloCtxCpu is a headless equivalent for unit tests (no GPU required).

// --- Draw command enum (shared by GPU and CPU paths) ----------------------

#[derive(Debug, Clone)]
pub enum DrawCmd {
    FillRect { rect: [f32; 4], color: [f32; 4] },
}

// --- CPU test stub --------------------------------------------------------

/// Headless VelloCtx for unit tests — accumulates DrawCmd without a GPU.
pub struct VelloCtxCpu {
    commands: Vec<DrawCmd>,
    width:    u32,
    height:   u32,
}

impl VelloCtxCpu {
    pub fn new(width: u32, height: u32) -> Self {
        Self { commands: Vec::new(), width, height }
    }

    pub fn begin_frame(&mut self) {
        self.commands.clear();
    }

    pub fn fill_rect(&mut self, rect: [f32; 4], color: [f32; 4]) {
        self.commands.push(DrawCmd::FillRect { rect, color });
    }

    pub fn commands(&self) -> &[DrawCmd] {
        &self.commands
    }

    pub fn resize(&mut self, width: u32, height: u32) {
        self.width  = width;
        self.height = height;
    }

    pub fn width(&self)  -> u32 { self.width  }
    pub fn height(&self) -> u32 { self.height }
}

// --- GPU VelloCtx (feature-gated) ----------------------------------------

#[cfg(feature = "game-ui")]
pub struct VelloCtx {
    renderer: vello::Renderer,
    scene:    vello::Scene,
    width:    u32,
    height:   u32,
}

#[cfg(feature = "game-ui")]
impl VelloCtx {
    /// Create from an existing wgpu Device/Queue.
    /// The surface texture format must match what vello expects (Bgra8Unorm recommended).
    pub fn new(
        device: &wgpu::Device,
        queue:  &wgpu::Queue,
        width:  u32,
        height: u32,
        surface_format: wgpu::TextureFormat,
    ) -> Result<Self, vello::Error> {
        let renderer = vello::Renderer::new(
            device,
            vello::RendererOptions {
                surface_format: Some(surface_format),
                use_cpu:        false,
                antialiasing_support: vello::AaSupport::area_only(),
                num_init_threads: std::num::NonZeroUsize::new(1),
            },
        )?;
        Ok(Self { renderer, scene: vello::Scene::new(), width, height })
    }

    pub fn begin_frame(&mut self) {
        self.scene = vello::Scene::new();
    }

    /// Draw a filled rectangle.
    /// `rect` = [x, y, width, height] in logical pixels.
    /// `color` = [r, g, b, a] in linear sRGB [0, 1].
    pub fn fill_rect(&mut self, rect: [f32; 4], color: [f32; 4]) {
        use vello::kurbo::{Affine, Rect};
        use vello::peniko::{Brush, Color, Fill};
        let vello_color = Color::rgba(
            color[0] as f64,
            color[1] as f64,
            color[2] as f64,
            color[3] as f64,
        );
        let vello_rect = Rect::new(
            rect[0] as f64,
            rect[1] as f64,
            (rect[0] + rect[2]) as f64,
            (rect[1] + rect[3]) as f64,
        );
        self.scene.fill(
            Fill::NonZero,
            Affine::IDENTITY,
            &Brush::Solid(vello_color),
            None,
            &vello_rect,
        );
    }

    /// Render the accumulated scene into a wgpu surface texture view.
    pub fn end_frame(
        &mut self,
        device: &wgpu::Device,
        queue:  &wgpu::Queue,
        surface_view: &wgpu::TextureView,
    ) -> Result<(), vello::Error> {
        self.renderer.render_to_texture(
            device,
            queue,
            &self.scene,
            surface_view,
            &vello::RenderParams {
                base_color:          vello::peniko::Color::BLACK,
                width:               self.width,
                height:              self.height,
                antialiasing_method: vello::AaConfig::Area,
            },
        )
    }

    pub fn resize(&mut self, width: u32, height: u32) {
        self.width  = width;
        self.height = height;
    }
}
```

```rust
// crates/vox_ui/src/spectral_hud.rs
//! SpectralHUD — live 16-band spectral energy bar display.
//!
//! render_cpu() uses VelloCtxCpu (tests / headless).
//! render()     uses VelloCtx (GPU, feature = "game-ui").
//!
//! Bar colors map spectral bands to approximate visible wavelengths:
//!   Band 0  (380–405 nm) → violet       [0.58, 0.00, 0.83]
//!   Band 1  (405–430 nm) → violet-blue  [0.35, 0.00, 1.00]
//!   Band 2  (430–455 nm) → blue-violet  [0.00, 0.00, 1.00]
//!   Band 3  (455–480 nm) → blue         [0.00, 0.35, 1.00]
//!   Band 4  (480–505 nm) → blue-cyan    [0.00, 0.60, 1.00]
//!   Band 5  (505–530 nm) → cyan         [0.00, 0.85, 0.85]
//!   Band 6  (530–555 nm) → cyan-green   [0.00, 0.90, 0.40]
//!   Band 7  (555–580 nm) → green-yellow [0.30, 0.90, 0.00]
//!   Band 8  (580–605 nm) → yellow       [1.00, 0.95, 0.00]
//!   Band 9  (605–630 nm) → yellow-orange[1.00, 0.75, 0.00]
//!   Band 10 (630–655 nm) → orange       [1.00, 0.55, 0.00]
//!   Band 11 (655–680 nm) → orange-red   [1.00, 0.30, 0.00]
//!   Band 12 (680–705 nm) → red          [1.00, 0.10, 0.00]
//!   Band 13 (705–730 nm) → deep red     [0.90, 0.00, 0.00]
//!   Band 14 (730–755 nm) → near-IR/red  [0.80, 0.00, 0.00]
//!   Band 15 (755–780 nm) → near-IR      [0.70, 0.00, 0.00]

use crate::vello_ctx::VelloCtxCpu;

/// Minimal spectral radiance cache for the UI layer.
/// Production code replaces this with a reference to vox_render::SpectralRadianceCache.
#[derive(Debug, Clone, Copy)]
pub struct SpectralRadianceCache {
    /// Per-band radiance energy, linear [0, 1].
    pub band_energy: [f32; 16],
}

impl Default for SpectralRadianceCache {
    fn default() -> Self { Self { band_energy: [0.0; 16] } }
}

pub struct SpectralHUD;

/// Visible-spectrum colour per band, [r, g, b, a].
const BAND_COLORS: [[f32; 4]; 16] = [
    [0.58, 0.00, 0.83, 1.0], // Band 0  — 380–405 nm — violet
    [0.35, 0.00, 1.00, 1.0], // Band 1  — 405–430 nm — violet-blue
    [0.00, 0.00, 1.00, 1.0], // Band 2  — 430–455 nm — blue-violet
    [0.00, 0.35, 1.00, 1.0], // Band 3  — 455–480 nm — blue
    [0.00, 0.60, 1.00, 1.0], // Band 4  — 480–505 nm — blue-cyan
    [0.00, 0.85, 0.85, 1.0], // Band 5  — 505–530 nm — cyan
    [0.00, 0.90, 0.40, 1.0], // Band 6  — 530–555 nm — cyan-green
    [0.30, 0.90, 0.00, 1.0], // Band 7  — 555–580 nm — green-yellow
    [1.00, 0.95, 0.00, 1.0], // Band 8  — 580–605 nm — yellow
    [1.00, 0.75, 0.00, 1.0], // Band 9  — 605–630 nm — yellow-orange
    [1.00, 0.55, 0.00, 1.0], // Band 10 — 630–655 nm — orange
    [1.00, 0.30, 0.00, 1.0], // Band 11 — 655–680 nm — orange-red
    [1.00, 0.10, 0.00, 1.0], // Band 12 — 680–705 nm — red
    [0.90, 0.00, 0.00, 1.0], // Band 13 — 705–730 nm — deep red
    [0.80, 0.00, 0.00, 1.0], // Band 14 — 730–755 nm — near-IR/red
    [0.70, 0.00, 0.00, 1.0], // Band 15 — 755–780 nm — near-IR
];

/// Bar layout constants.
const BAR_WIDTH:  f32 = 8.0;
const BAR_GAP:    f32 = 2.0;

impl SpectralHUD {
    /// Returns per-band colors (violet → near-IR gradient).
    pub fn band_colors() -> [[f32; 4]; 16] {
        BAND_COLORS
    }

    /// Compute bar rects from energy values.
    ///
    /// - `energy`: per-band energy [0, 1].
    /// - `pos`: top-left corner [x, y] of the HUD in logical pixels.
    /// - `total_width`: total width of the 16-bar group.
    /// - `max_height`: maximum bar height (for energy = 1.0).
    ///
    /// Returns 16 rects as [x, y, width, height].
    /// Bars grow upward from `pos.y + max_height`.
    pub fn bar_rects(
        energy:      [f32; 16],
        pos:         [f32; 2],
        total_width: f32,
        max_height:  f32,
    ) -> Vec<[f32; 4]> {
        let bar_w = (total_width / 16.0 - BAR_GAP).max(1.0);
        (0..16usize).map(|b| {
            let h   = (energy[b].clamp(0.0, 1.0) * max_height).max(0.0);
            let x   = pos[0] + b as f32 * (bar_w + BAR_GAP);
            let y   = pos[1] + (max_height - h); // grow upward
            [x, y, bar_w, h]
        }).collect()
    }

    /// Render the SpectralHUD into a CPU (headless) VelloCtxCpu.
    /// Used in tests and headless benchmarks.
    pub fn render_cpu(
        ctx:   &mut VelloCtxCpu,
        cache: &SpectralRadianceCache,
        pos:   [f32; 2],
    ) {
        let max_height  = 60.0;
        let total_width = 160.0;
        let bars        = Self::bar_rects(cache.band_energy, pos, total_width, max_height);
        let colors      = Self::band_colors();

        for (b, rect) in bars.iter().enumerate() {
            // Background track (dim, full height)
            let bg_rect = [rect[0], pos[1], rect[2], max_height];
            ctx.fill_rect(bg_rect, [0.1, 0.1, 0.1, 0.7]);
            // Energy bar
            ctx.fill_rect(*rect, colors[b]);
        }
    }

    /// Render the SpectralHUD into a GPU VelloCtx.
    /// `pos` = top-left corner in logical pixels.
    #[cfg(feature = "game-ui")]
    pub fn render(
        ctx:   &mut crate::vello_ctx::VelloCtx,
        cache: &SpectralRadianceCache,
        pos:   [f32; 2],
    ) {
```

```rust
// crates/vox_ui/src/layout.rs
//! Taffy-backed flexbox layout for vox_ui game widgets.
//!
//! LayoutTree owns a TaffyTree and resolves widget rects each frame.
//! Gate under feature = "game-ui" since taffy is optional.
//!
//! Without the feature, a stub implementation compiles so dependent
//! code in lib.rs can be feature-gated cleanly.

#[cfg(feature = "game-ui")]
use taffy::prelude::*;

#[cfg(feature = "game-ui")]
pub type LayoutNodeId = NodeId;

#[cfg(not(feature = "game-ui"))]
pub type LayoutNodeId = u64;

#[cfg(feature = "game-ui")]
pub struct LayoutTree {
    tree: TaffyTree,
}

#[cfg(feature = "game-ui")]
impl LayoutTree {
    pub fn new() -> Self {
        Self { tree: TaffyTree::new() }
    }

    /// Create a root row-flex container with a fixed pixel size.
    /// Uses `new_with_children` (not `new_leaf`) because children will be added later;
    /// in taffy 0.7, leaf nodes cannot have children.
    pub fn add_row_container(&mut self, width: f32, height: f32) -> LayoutNodeId {
        self.tree.new_with_children(Style {
            display:   Display::Flex,
            flex_direction: FlexDirection::Row,
            size: Size {
                width:  Dimension::Length(width),
                height: Dimension::Length(height),
            },
            ..Default::default()
        }, &[]).expect("taffy container")
    }

    /// Add a flex child that grows proportionally to `flex_grow`.
    pub fn add_flex_child(&mut self, parent: LayoutNodeId, flex_grow: f32) -> LayoutNodeId {
        let child = self.tree.new_leaf(Style {
            flex_grow,
            size: Size { width: Dimension::Auto, height: Dimension::Auto },
            ..Default::default()
        }).expect("taffy child leaf");
        self.tree.add_child(parent, child).expect("add child");
        child
    }

    /// Resolve layout for the entire tree given a viewport size.
    pub fn resolve(&mut self, viewport_width: f32, viewport_height: f32) {
        self.tree.compute_layout_with_measure(
            // Taffy needs a root node; iterate all roots.
            // For simplicity: compute for each root-level node.
            // Production: pass explicit root node.
            *self.tree.children(
                self.tree.children(taffy::tree::NodeId::from(0u64))
                    .ok().and_then(|c| c.first().copied())
                    .unwrap_or(taffy::tree::NodeId::from(0u64))
            ).ok().and_then(|c| c.first()).unwrap_or(
                &taffy::tree::NodeId::from(0u64)
            ),
            taffy::geometry::Size {
                width:  AvailableSpace::Definite(viewport_width),
                height: AvailableSpace::Definite(viewport_height),
            },
            |_, _| Ok(taffy::geometry::Size::ZERO),
        ).ok();
    }

    /// Resolved rect for a node: [x, y, width, height] in logical pixels.
    pub fn rect(&self, node: LayoutNodeId) -> Option<[f32; 4]> {
        let layout = self.tree.layout(node).ok()?;
        Some([
            layout.location.x,
            layout.location.y,
            layout.size.width,
            layout.size.height,
        ])
    }
}

#[cfg(not(feature = "game-ui"))]
pub struct LayoutTree;

#[cfg(not(feature = "game-ui"))]
impl LayoutTree {
    pub fn new() -> Self { Self }
    pub fn add_row_container(&mut self, _w: f32, _h: f32) -> LayoutNodeId { 0 }
    pub fn add_flex_child(&mut self, _parent: LayoutNodeId, _flex_grow: f32) -> LayoutNodeId { 0 }
    pub fn resolve(&mut self, _w: f32, _h: f32) {}
    pub fn rect(&self, _node: LayoutNodeId) -> Option<[f32; 4]> { Some([0.0; 4]) }
}
```

Note: the Taffy resolve loop above is illustrative — consult Taffy 0.7 docs to confirm the correct `compute_layout` call signature and root node traversal API (Taffy's API evolved significantly between 0.3 and 0.7). Replace with a direct `tree.compute_layout(root, available_space)` call once the root NodeId is stored explicitly.
